add_subdirectory(external/googletest-1.16.0) # add_subdirectory -> CMakeLists.txt 서브프로젝트를 감지하는 명령
add_executable(<빌드 타겟> <빌드대상 소스>) # -> 빌드할 소스코드 파일을 'EXE로 만들 타겟' 객체에 추가하는 명령
add_library(<라이브러리 타겟> <라이브러리 넣을 소스>) # -> 단위 소스코드 파일을 '라이브러리 단위로 묶을 타겟' 객체에 추가하는 명령
target_link_libraries(<대상 타겟> <소스 타겟>) # -> '소스 타겟' 을 '대상 타겟' 에 추가하는 명령

   <Library Source>
        ㄴ add_library  -> <Library Target>

   <EXE Source>
       ㄴ add_executable -> <EXE Target>

    <Library Target>
       ㄴ target_link_libraries -> <EXE Target>

	target_link_options(<테스트 EXE 타겟> PRIVATE --coverage) # -> -> (링커용) 해당 타겟에 coverage 정보를 포함시키는 명령

	target_link_libraries(person_test PRIVATE --coverage) # (구버전, 링커용) 해당 타겟에 (다른 타겟 대신) coverage 정보를 포함시키는 명령 (target_link_options 와 교체관계)
